(function () {

// ----------------------------------------------------------------------------- // ----------------------------------------------------------------------------- // -- // -- File Name: app.js // -- // -- Author: Paul Nispel // -- // ----------------------------------------------------------------------------- // -- Copyright 2014 Skwintz // -----------------------------------------------------------------------------

var module = angular.module( 'directive.bind_once', []);
module.directive('bindOnce', function() { return { scope: true, link: function( $scope ) { setTimeout(function() { $scope.$destroy(); }, 0); } }; });
})();
